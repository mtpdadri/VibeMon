📱 Mobile Usage — Important
Why data does not load when opening .html file directly on phone:
Chrome on Android blocks all network requests (fetch/CORS) when opening a local file:// file — this prevents Google Sheet connection.

✅ Correct way to use VibeMon on mobile:
1. Host this file on Google Drive → share as "Anyone with link" → open that link in Chrome.
   OR
2. Host on GitHub Pages (free) → open the https:// link on mobile.
   OR
3. Run a simple local server: in the folder containing this file, run python -m http.server 8080 and open http://YOUR-PC-IP:8080 on phone (same WiFi).

💡 Quickest option: Upload this .html to Google Drive → right-click → Get link → Anyone with link → share that link with all users.

var TZ = Session.getScriptTimeZone();

function fmtDate(d) {
  return d instanceof Date ? Utilities.formatDate(d, TZ, 'yyyy-MM-dd') : String(d || '');
}

function fmtDateTime(d) {
  return d instanceof Date ? Utilities.formatDate(d, TZ, 'yyyy-MM-dd HH:mm:ss') : String(d || '');
}

function ensureAllReadingsHeader(sh) {
  if (sh.getLastRow() === 0) {
    sh.appendRow(['Report_ID','Date','Username','Inspector','Unit','Area','Equipment','RPM','Frequency','Point','H Vel','V Vel','A Vel','Acc','H Dis','V Dis','A Dis','Severity','Recommendation','Remarks','Created_On','Updated_On']);
  }
}

function generateReportId(dateStr) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName('ALL_READINGS');
  var d = dateStr.replace(/-/g,'');
  var prefix = 'VBM-' + d + '-';
  var maxNum = 0;
  if (sh && sh.getLastRow() > 1) {
    var vals = sh.getRange(2, 1, sh.getLastRow()-1, 1).getValues();
    vals.forEach(function(r) {
      var id = String(r[0] || '');
      if (id.indexOf(prefix) === 0) {
        var num = parseInt(id.replace(prefix,''), 10);
        if (!isNaN(num) && num > maxNum) maxNum = num;
      }
    });
  }
  return prefix + ('000' + (maxNum + 1)).slice(-4);
}

// ─── doPost ───────────────────────────────────────────────────────────────────
function doPost(e) {
  try {
    var raw = (e.parameter && e.parameter.payload) ? e.parameter.payload : e.postData.contents;
    var data = JSON.parse(raw);
    var action = data.action || 'addRows';
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var allSh = ss.getSheetByName('ALL_READINGS') || ss.insertSheet('ALL_READINGS');
    ensureAllReadingsHeader(allSh);

    // ── loginUser ──────────────────────────────────────────────────────────────
    if (action === 'loginUser') {
      var uname = String(data.username || '').trim();
      var pass  = String(data.password || '');
      var uSh   = ss.getSheetByName('USERS');
      if (!uSh || uSh.getLastRow() <= 1) {
        return json({status:'error', msg:'USERS sheet not found or empty.'});
      }
      var uData = uSh.getDataRange().getValues();
      // cols: Username(0) Password(1) Role(2) Name(3) Status(4)
      for (var i = 1; i < uData.length; i++) {
        if (String(uData[i][0]).trim().toLowerCase() === uname.toLowerCase() &&
            String(uData[i][1]) === pass &&
            String(uData[i][4] || 'ACTIVE').toUpperCase() === 'ACTIVE') {
          return json({status:'ok', user:{
            username: String(uData[i][0]).trim(),
            password: '', // never send back
            role:     String(uData[i][2]).trim().toUpperCase(),
            name:     String(uData[i][3]).trim()
          }});
        }
      }
      return json({status:'error', msg:'Invalid credentials or inactive account.'});
    }

    // ── saveReport ─────────────────────────────────────────────────────────────
    if (action === 'saveReport') {
      var reportId  = data.reportId || generateReportId(data.date || new Date().toISOString().slice(0,10));
      var now       = fmtDateTime(new Date());
      var readings  = data.readings || [];
      var recommendation = String(data.recommendation || '');
      var remarks   = String(data.remarks || '');

      // Build rows batch
      var newRows = readings.map(function(pt) {
        return [
          reportId,
          data.date || '',
          data.username || '',
          data.inspector || '',
          data.unit || '',
          data.area || '',
          data.equipment || '',
          data.rpm || '',
          data.frequency || '',
          pt.point || '',
          pt.H_Vel !== undefined ? pt.H_Vel : '',
          pt.V_Vel !== undefined ? pt.V_Vel : '',
          pt.A_Vel !== undefined ? pt.A_Vel : '',
          pt.Acc   !== undefined ? pt.Acc   : '',
          pt.H_Dis !== undefined ? pt.H_Dis : '',
          pt.V_Dis !== undefined ? pt.V_Dis : '',
          pt.A_Dis !== undefined ? pt.A_Dis : '',
          pt.severity || '',
          recommendation,
          remarks,
          now,
          now
        ];
      });

      if (newRows.length > 0) {
        var startRow = allSh.getLastRow() + 1;
        allSh.getRange(startRow, 1, newRows.length, 22).setValues(newRows);
      }
      return json({status:'ok', reportId: reportId});
    }

    // ── updateReport ───────────────────────────────────────────────────────────
    if (action === 'updateReport') {
      var updId = data.reportId || '';
      var now2  = fmtDateTime(new Date());
      if (!updId) return json({status:'error', msg:'reportId required'});
      var lastRow = allSh.getLastRow();
      if (lastRow <= 1) return json({status:'ok', updated:0});
      var vals = allSh.getRange(2, 1, lastRow-1, 22).getValues();
      var readings2 = data.readings || [];
      // Map point → new data
      var ptMap = {};
      readings2.forEach(function(pt){ ptMap[pt.point] = pt; });
      var updCount = 0;
      for (var ri = 0; ri < vals.length; ri++) {
        if (String(vals[ri][0]) !== updId) continue;
        var pt = String(vals[ri][9]);
        var newPtData = ptMap[pt];
        var rowNum = ri + 2;
        // Update recommendation and remarks on all rows of this report
        if (data.recommendation !== undefined) allSh.getRange(rowNum, 19).setValue(data.recommendation);
        if (data.remarks        !== undefined) allSh.getRange(rowNum, 20).setValue(data.remarks);
        if (newPtData) {
          // Update measurement values and severity
          if (newPtData.H_Vel !== undefined) allSh.getRange(rowNum, 11).setValue(newPtData.H_Vel);
          if (newPtData.V_Vel !== undefined) allSh.getRange(rowNum, 12).setValue(newPtData.V_Vel);
          if (newPtData.A_Vel !== undefined) allSh.getRange(rowNum, 13).setValue(newPtData.A_Vel);
          if (newPtData.Acc   !== undefined) allSh.getRange(rowNum, 14).setValue(newPtData.Acc);
          if (newPtData.H_Dis !== undefined) allSh.getRange(rowNum, 15).setValue(newPtData.H_Dis);
          if (newPtData.V_Dis !== undefined) allSh.getRange(rowNum, 16).setValue(newPtData.V_Dis);
          if (newPtData.A_Dis !== undefined) allSh.getRange(rowNum, 17).setValue(newPtData.A_Dis);
          if (newPtData.severity) allSh.getRange(rowNum, 18).setValue(newPtData.severity);
        }
        allSh.getRange(rowNum, 22).setValue(now2);
        updCount++;
      }
      return json({status:'ok', updated: updCount});
    }

    // ── deleteReport ───────────────────────────────────────────────────────────
    if (action === 'deleteReport') {
      var delId   = data.reportId || '';
      var delDate = String(data.date || '');
      var delEquip= String(data.equipment || '');
      var lastRow3= allSh.getLastRow();
      if (lastRow3 > 1) {
        var d3 = allSh.getRange(2, 1, lastRow3-1, 8).getValues();
        for (var i3 = d3.length-1; i3 >= 0; i3--) {
          var rowDate3 = fmtDate(d3[i3][1]);
          var matchId   = delId   && String(d3[i3][0]) === delId;
          var matchDateEquip = !delId && rowDate3 === delDate && String(d3[i3][6]) === delEquip;
          if (matchId || matchDateEquip) allSh.deleteRow(i3 + 2);
        }
      }
      return json({status:'ok', deleted:true});
    }

    // ── legacy deleteRows (backward compat) ────────────────────────────────────
    if (action === 'deleteRows') {
      var dDate = String(data.date||''), dEquip = String(data.equipment||'');
      var lr = allSh.getLastRow();
      if (lr > 1) {
        var dd = allSh.getRange(2,1,lr-1,22).getValues();
        for (var ix=dd.length-1;ix>=0;ix--) {
          var rd=fmtDate(dd[ix][1]||dd[ix][0]);
          var eq=String(dd[ix][6]||dd[ix][1]);
          if(rd===dDate && eq===dEquip) allSh.deleteRow(ix+2);
        }
      }
      return json({status:'ok',deleted:true});
    }

    // ── legacy editRows / addRows ──────────────────────────────────────────────
    if (action === 'editRows' || action === 'addRows') {
      var rows = data.rows || [];
      var now3 = fmtDateTime(new Date());
      if (action === 'editRows' && rows.length > 0) {
        var eDate = rows[0].date, eEquip = rows[0].equipment;
        var lr2 = allSh.getLastRow();
        if (lr2 > 1) {
          var ed = allSh.getRange(2,1,lr2-1,22).getValues();
          for (var ix2=ed.length-1;ix2>=0;ix2--) {
            var rd2=fmtDate(ed[ix2][1]||ed[ix2][0]);
            var eq2=String(ed[ix2][6]||ed[ix2][1]);
            if(rd2===eDate && eq2===eEquip) allSh.deleteRow(ix2+2);
          }
        }
      }
      var existing = {};
      if (allSh.getLastRow() > 1) {
        var ev = allSh.getRange(2,1,allSh.getLastRow()-1,22).getValues();
        ev.forEach(function(r,i){ existing[fmtDate(r[1]||r[0])+'||'+String(r[6]||r[1])+'||'+String(r[9]||r[4])]=i+2; });
      }
      rows.forEach(function(r) {
        var k=(r.date)+'||'+(r.equipment)+'||'+(r.point);
        var rid = r.reportId || '';
        var row22 = [rid,r.date,r.username||r.inspector||'',r.inspector||'',r.unit,r.area,r.equipment,r.rpm||'',r.frequency||'',r.point,r.H_Vel,r.V_Vel,r.A_Vel,r.Acc,r.H_Dis,r.V_Dis,r.A_Dis,r.severity,r.recommendations||r.recommendation||'',r.remarks||'',now3,now3];
        if(!existing[k]){
          allSh.appendRow(row22);
        } else {
          allSh.getRange(existing[k],1,1,22).setValues([row22]);
        }
      });
      return json({status:'ok'});
    }

    return json({status:'error', msg:'Unknown action: ' + action});
  } catch(err) {
    return json({status:'error', msg: err.message + ' | ' + err.stack});
  }
}

// ─── doGet ────────────────────────────────────────────────────────────────────
function doGet(e) {
  try {
    var action = (e && e.parameter) ? (e.parameter.action || '') : '';
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    // ── ping ──────────────────────────────────────────────────────────────────
    if (action === 'ping') return json({status:'ok', msg:'VibeMon v4 active'});

    // ── loginUser (GET) ───────────────────────────────────────────────────────
    if (action === 'loginUser') {
      var uname = String((e && e.parameter) ? (e.parameter.username || '') : '').trim();
      var pass  = String((e && e.parameter) ? (e.parameter.password || '') : '');
      var uSh   = ss.getSheetByName('USERS');
      if (!uSh || uSh.getLastRow() <= 1) {
        return json({status:'error', msg:'USERS sheet not found or empty.'});
      }
      var uData = uSh.getDataRange().getValues();
      for (var i = 1; i < uData.length; i++) {
        if (String(uData[i][0]).trim().toLowerCase() === uname.toLowerCase() &&
            String(uData[i][1]) === pass &&
            String(uData[i][4] || 'ACTIVE').toUpperCase() === 'ACTIVE') {
          return json({status:'ok', user:{
            username: String(uData[i][0]).trim(),
            password: '',
            role:     String(uData[i][2]).trim().toUpperCase(),
            name:     String(uData[i][3]).trim()
          }});
        }
      }
      return json({status:'error', msg:'Invalid credentials or inactive account.'});
    }

    // ── getAll / getReadings ───────────────────────────────────────────────────
    if (action === 'getAll' || action === 'getReadings') {
      var allSh = ss.getSheetByName('ALL_READINGS');
      if (!allSh || allSh.getLastRow() <= 1) return json({status:'ok', rows:[]});
      var d = allSh.getDataRange().getValues();
      var rows = [];
      // Support both old (15-col) and new (22-col) format
      for (var i = 1; i < d.length; i++) {
        var r = d[i]; if (!r[0] && !r[1]) continue;
        var is22 = d[0].length >= 22;
        if (is22) {
          rows.push({
            reportId: String(r[0]||''),
            date:     fmtDate(r[1]),
            username: String(r[2]||''),
            inspector:String(r[3]||''),
            unit:     String(r[4]||''),
            area:     String(r[5]||''),
            equipment:String(r[6]||''),
            rpm:      String(r[7]||''),
            frequency:String(r[8]||''),
            point:    String(r[9]||''),
            H_Vel:    String(r[10]||''), V_Vel:String(r[11]||''), A_Vel:String(r[12]||''),
            Acc:      String(r[13]||''), H_Dis:String(r[14]||''), V_Dis:String(r[15]||''), A_Dis:String(r[16]||''),
            severity: String(r[17]||'NORMAL'),
            recommendations: String(r[18]||''),
            remarks:  String(r[19]||''),
            createdOn:String(r[20]||''),
            updatedOn:String(r[21]||'')
          });
        } else {
          // legacy 15-col format
          rows.push({
            reportId:'', date:fmtDate(r[0]), username:'', inspector:String(r[13]||''),
            unit:String(r[2]||''), area:String(r[3]||''), equipment:String(r[1]||''),
            rpm:'', frequency:'', point:String(r[4]||''),
            H_Vel:String(r[5]||''),V_Vel:String(r[6]||''),A_Vel:String(r[7]||''),
            Acc:String(r[8]||''),H_Dis:String(r[9]||''),V_Dis:String(r[10]||''),A_Dis:String(r[11]||''),
            severity:String(r[12]||'NORMAL'), recommendations:String(r[14]||''),
            remarks:'', createdOn:'', updatedOn:''
          });
        }
      }
      return json({status:'ok', rows:rows});
    }

    // ── getEquipmentMaster ─────────────────────────────────────────────────────
    if (action === 'getEquipmentMaster') {
      var emSh = ss.getSheetByName('Equipment Master') || ss.getSheetByName('EQUIPMENT MASTER') || ss.getSheetByName('equipment master');
      if (!emSh || emSh.getLastRow() <= 1) return json({status:'ok', equipment:[]});
      var em = emSh.getDataRange().getValues();
      // Header: S.NO.(0) UNITS/STAGE(1) AREA(2) EQUIPMENTS NAME(3) KW(4) RPM(5) VIBRATION LIMITS(6) EQUIPMENT ROUTES(7) Frequency(8) PARAMETERS(9)
      var equipment = [];
      for (var i = 1; i < em.length; i++) {
        var r = em[i];
        if (!r[3]) continue; // skip empty rows
        var limStr = String(r[6]||'');
        var limits = parseLimits(limStr);
        equipment.push({
          sno:       String(r[0]||''),
          unit:      String(r[1]||'').trim(),
          area:      String(r[2]||'').trim(),
          name:      String(r[3]||'').trim(),
          kw:        String(r[4]||''),
          rpm:       String(r[5]||''),
          limitsRaw: limStr,
          limits:    limits,
          points:    String(r[7]||'').split(',').map(function(s){return s.trim();}).filter(Boolean),
          frequency: parseInt(r[8]||'15', 10) || 15,
          params:    String(r[9]||'H Vel,V Vel,A Vel,Acc').split(',').map(function(s){return s.trim();}).filter(Boolean)
        });
      }
      return json({status:'ok', equipment:equipment});
    }

    // ── getUsers ───────────────────────────────────────────────────────────────
    if (action === 'getUsers') {
      var uSh2 = ss.getSheetByName('USERS');
      if (!uSh2 || uSh2.getLastRow() <= 1) return json({status:'ok', users:[]});
      var ud = uSh2.getDataRange().getValues();
      var users = [];
      for (var i = 1; i < ud.length; i++) {
        if (String(ud[i][4]||'ACTIVE').toUpperCase() === 'ACTIVE') {
          users.push({ username: String(ud[i][0]).trim(), role: String(ud[i][2]).trim().toUpperCase(), name: String(ud[i][3]).trim() });
        }
      }
      return json({status:'ok', users:users});
    }

    // ── getRecommendations ────────────────────────────────────────────────────
    if (action === 'getRecommendations') {
      var recSh = ss.getSheetByName('RECOMMENDATIONS_MASTER');
      if (!recSh || recSh.getLastRow() <= 1) return json({status:'ok', recommendations:[]});
      var rd = recSh.getDataRange().getValues();
      var recs = [];
      for (var i = 1; i < rd.length; i++) {
        if (rd[i][1]) recs.push({ category: String(rd[i][0]||'General'), recommendation: String(rd[i][1]) });
      }
      return json({status:'ok', recommendations:recs});
    }

    return json({status:'ok', msg:'VibeMon v4 active'});
  } catch(err) {
    return json({status:'error', msg: err.message});
  }
}

// ─── Helpers ──────────────────────────────────────────────────────────────────
function json(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

function parseLimits(limStr) {
  // Parse strings like:
  //  "Normal Upto 4.5 mm/s, Alarm 4.5 to 7.1 mm/s, Alert 7.1 to 11.2 mm/s, Critical >11.2 mm/s"
  //  "Normal Up to 50 mic, Alarm 50 to 75 mic, Alert 75 to 100 mic, Critical >100 mic"
  if (!limStr) return null;
  var s = limStr.replace(/\n/g,' ').replace(/,/g,' ');
  var unit = (s.toLowerCase().indexOf('mic') >= 0) ? 'mic' : 'mm/s';
  function extract(pat) {
    var m = s.match(pat);
    return m ? parseFloat(m[1]) : null;
  }
  var alarm    = extract(/alarm[\s\S]*?(\d+(?:\.\d+)?)\s*(?:to|mm|mic)/i);
  var alertVal = extract(/alert[\s\S]*?(\d+(?:\.\d+)?)\s*(?:to|mm|mic)/i);
  var crit     = extract(/critical\s*[>≥]\s*(\d+(?:\.\d+)?)/i);
  if (!alarm) alarm    = extract(/(\d+(?:\.\d+)?)\s*to\s*\d/i);
  if (!crit)  crit     = extract(/>(\d+(?:\.\d+)?)/i);
  if (alarm === null && alertVal === null && crit === null) return null;
  return { u: unit, a: alarm || 4.5, at: alertVal || 7.1, c: crit || 11.2 };
}
